orderList.test.ts — AI 自动生成 
// AI 根据组件源码 + 接口文档自动生成的测试用例
import { render, screen, waitFor } from '@testing-library/react';
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import OrderList from './OrderList';
import { server } from './mocks/server';

// AI 自动推导：解析 OrderList 组件 → 识别状态 → 生成边界 case
describe('OrderList Component', () => {
  let queryClient: QueryClient;

  beforeEach(() => {
    queryClient = new QueryClient({
      defaultOptions: { queries: { retry: false } },
    });
  });

  // ✓ 空状态
  it('renders empty state when no orders', async () => {
    server.use('...empty-orders-mock');
    render(
      new QueryClientProvider({ client: queryClient },
        <OrderList />
      )
    );
    expect(screen.getByText('暂无订单')).toBeInTheDocument();
  });

  // ✓ 加载中状态
  it('shows skeleton while loading', () => {
    render(<OrderList />);
    expect(document.querySelector('.skeleton')).toBeInTheDocument();
  });

  // ✓ 错误状态
  it('displays error message on fetch failure', async () => {
    server.use('...error-mock');
    render(<OrderList />);
    await waitFor(() =>
      expect(screen.getByText('加载失败')).toBeInTheDocument()
    );
  });

  // ✓ 数据渲染
  it('renders order items correctly', async () => {
    render(<OrderList />);
    await waitFor(() => expect(screen.getByText('CS2 AWP')).toBeInTheDocument());
  });
});
📊 成果：测试覆盖率 从 ~30% → 70%，AI 自动生成测试用例，工程师只需补充边界断言
